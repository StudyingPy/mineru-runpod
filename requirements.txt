# Container-side dependencies. Consumers do NOT need these — they only need
# the mineru_client subpackage which has its own pyproject.toml.

# MinerU with vLLM backend support. The 3.x line is where the library default
# became opendatalab/MinerU2.5-Pro-2604-1.2B (the model this template name
# advertises). 3.1.x requires vllm 0.10.1.1+ — see Dockerfile FROM tag.
# Pinned to a 3.1.x line so the handler behaviour stays stable; bump deliberately.
mineru[core,vllm]>=3.1.0,<4.0.0

# RunPod serverless SDK — provides runpod.serverless.start() and async helpers.
runpod>=1.7.0

# Used by the handler for downloading URL-supplied PDFs.
httpx>=0.27

# Used by the handler when callers request `return: "s3"` to upload the
# output tarball to an S3-compatible bucket and return a presigned URL.
# Only imported lazily inside the s3 code path, so a worker without S3
# env vars never touches it.
boto3>=1.35

# Xet storage backend client used by huggingface_hub for chunk-based
# parallel downloads. Auto-detected when installed; the Dockerfile bake
# step sets HF_XET_HIGH_PERFORMANCE=1 to saturate bandwidth and CPU
# during the model snapshot pull. Replaces the deprecated hf_transfer.
# Runtime stays offline so this is build-time only.
hf-xet>=1.0
